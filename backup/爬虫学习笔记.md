import requests  
from bs4 import BeautifulSoup  
import pandas as pd  
  
ulist = []  

# 爬取的网站的URL  
url = "http://www.gaosan.com/gaokao/241219.html"  
response = requests.get(url)  

# 编码格式  
response.encoding = 'utf-8' 

# 编译数据  
soup = BeautifulSoup(response.text, 'html.parser')

'''
1. `ulist = []`：创建一个空列表，可能用于存储爬取的数据。
2. `url = "http://www.gaosan.com/gaokao/241219.html"`：设置要爬取的网页的 URL。
3. `response = requests.get(url)`：使用 `requests.get` 函数发送一个 GET 请求到指定的 URL，并将服务器的响应存储在 `response` 变量中。
4. `response.encoding = 'utf-8'`：设置 `response` 的编码格式为 'utf-8'。这是因为网页的内容通常是以文本形式发送的，需要指定一个编码格式来正确地解码这些文本。
5. `soup = BeautifulSoup(response.text, 'html.parser')`：使用 `BeautifulSoup` 函数解析 `response` 的文本内容。'html.parser' 是用于解析 HTML 的解析器。


BeautifulSoup 支持多种解析器，每种解析器都有其优点和缺点。以下是一些常用的解析器：
1. `'html.parser'`：这是一个 Python 标准库中的 HTML 解析器。它的优点是速度适中，容错性强。
2. `'lxml'`：这是一个非常高效的 XML 和 HTML 解析器。它的速度非常快，容错性也很好。但是，你需要额外安装 lxml 库才能使用它。
3. `'lxml-xml'` 或 `'xml'`：这是 lxml 库的 XML 解析器。它的速度非常快，但是容错性较差。
4. `'html5lib'`：这是一个解析 HTML5 的解析器。它的优点是容错性极强，能够解析非常混乱的 HTML 代码。但是，它的速度较慢，而且你需要额外安装 html5lib 库才能使用它。
你可以根据你的具体需求来选择合适的解析器。例如，如果你需要解析非常混乱的 HTML 代码，你可能会选择 'html5lib'。如果你需要解析速度非常快，你可能会选择 'lxml'。
'''

# 将数据存入定义好的ulist  
for tr in soup.find('tbody').children:  
   tds = tr('td')  
   ulist.append([tds[0].string, tds[1].string, tds[2].string, tds[3].string, tds[4].string, tds[5].string]) 
    
# 获取指定数量的大学并打印：  
   # {:^10}表示将元素居中对齐，宽度为10个字符。  
   # \t表示制表符，用于分隔不同的列。  
   # chr(12288)通常用于中文文本的排版，以保持整齐的对齐。
for i in range(21):  
   u = ulist[i]  
   print("{:^10}\t{:^20}\t{:^10}\t{:^10}\t{:^10}\t{:^10}".format(u[0], u[1], u[2], u[3], u[4], u[5], chr(12288))) 
    
# 创建一个pandas DataFrame对象  
df = pd.DataFrame(ulist[1:21], columns=ulist[0])  
  
# 将数据保存到Excel文件中  
df.to_excel('学校排名.xlsx', index=False)
